    • Configure [[L3 Services - IPv4 and HSRP#Border Gateway Protocol (BGP)|BGP]]. 
    en
conf t

ip prefix-list SUMMARY_VLAN seq 5 permit 10.10.100.0/22
ip prefix-list SUMMARY_VLAN seq 10 permit 10.11.100.0/22
route-map ISPV4_EXPORT permit 10
match ip address prefix-list SUMMARY_VLAN
set metric 100 # Lower than IR1
exit

# Learned routes from ISP BGP
ip prefix-list ISP_TRAFFIC seq 10 permit 0.0.0.0/0
route-map IR2_PREFER permit 10
match ip address prefix-list ISP_TRAFFIC
set local-preference 200 # Higher than IR1 default but lower than 198.51.100.0 and 203.0.113.0 prefixes
exit

router bgp 65000.3
bgp asnotation dot
bgp confederation identifier 65000
bgp confederation peers 65000.1

neighbor 192.0.2.201 remote-as 65000.1
neighbor 192.0.2.201 timers 10 30
neighbor 192.0.2.201 password Skill39@Manila
neighbor 192.0.2.201 route-map IR2_PREFER in
neighbor 192.0.2.201 route-map ISPV4_EXPORT out

neighbor WSA2025 peer-group
neighbor WSA2025 remote-as 65000.3
neighbor WSA2025 timers 10 30
neighbor WSA2025 password Skill39@Manila
neighbor WSA2025 update-source lo1
neighbor 10.10.0.1 peer-group WSA2025

network 10.10.100.0 mask 255.255.252.0
network 10.11.100.0 mask 255.255.252.0
aggregate-address 10.10.100.0 255.255.252.0 summary-only
aggregate-address 10.11.100.0 255.255.252.0 summary-only

end

    • Configure [[L3 Services - IPv4 and HSRP#Enhanced Interior Gateway Routing Protocol|EIGRP]]. Enable ipv6 unicast-routing. 
    en
conf t
ipv6 unicast-routing

router eigrp WSA2025
address-family ipv4 autonomous-system 100
eigrp router-id 10.10.0.11
network 10.10.0.11 0.0.0.0
network 10.10.0.204 0.0.0.3
network 10.10.0.208 0.0.0.3
network 172.16.100.0 0.0.0.255

topology base
# Advertise default and other routes received from ISP
redistribute bgp 65000.3 metric 1000000 1000 255 1 1500

af-interface default
passive-interface
exit
af-interface g0/1
no passive-interface
exit
af-interface g0/2
no passive-interface
exit
af-interface tun0
no passive-interface
# Allow re-advertising on the multi-point tunnel interface
no split-horizon 
# Allow spoke-to-spoke connectivity
no next-hop-self
exit
exit

address-family ipv6 autonomous-system 100
eigrp router-id 10.10.0.11

af-interface default
passive-interface
exit
af-interface g0/1
no passive-interface
exit
af-interface g0/2
no passive-interface
exit
af-interface tun0
no passive-interface

end

    • Configure NAT
    en
conf t

ip access-list standard VLAN101-102
permit 10.1.101.0 0.0.0.255
permit 10.1.102.0 0.0.0.255
permit 10.2.101.0 0.0.0.255
permit 10.2.102.0 0.0.0.255
permit 10.10.101.0 0.0.0.255
permit 10.10.102.0 0.0.0.255
permit 10.11.101.0 0.0.0.255
permit 10.11.102.0 0.0.0.255

ip nat pool IR2_POOL 192.0.2.193 192.0.2.199 netmask 255.255.255.240
ip nat inside source list VLAN101-102 pool IR2_POOL overload
ip nat inside source static 10.10.100.101 192.0.2.205

int g0/3
ip nat outside

int range g0/1-2,tun0
ip nat inside

end

    • Configure NHRP (Next Hop Redundancy Protocol).
    en
configure terminal

interface tun0
ip nhrp authentication WSA2025
ip nhrp network-id 2025
ip nhrp map multicast dynamic

# Phase 3
ip nhrp redirect
tunnel source g0/3
tunnel mode gre multipoint

end

    • Configure SSH
    ena
conf t

crypto key generate rsa modulus 2048
ip ssh version 2
ip ssh server algorithm mac hmac-sha2-256 hmac-sha2-512

# SSH ACL
ip access-list standard SSH_ACL
permit host 10.10.100.101

line vty 0 4
login local
transport input ssh
exec-timeout 0 0
access-class SSH_ACL in

end
