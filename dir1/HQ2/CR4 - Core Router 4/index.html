    • Configure [[L3 Services - IPv4 and HSRP#Open Shortest Path First (OSPF)|OSPF]].
    en
conf t

ip prefix-list HQ2-SUBNETS seq 5 permit 10.10.0.0/27 le 32
ip prefix-list HQ2-SUBNETS seq 10 permit 10.11.100.0/22 le 24
ip prefix-list HQ2-SUBNETS seq 15 permit 10.1.100.0/22 le 24
ip prefix-list HQ2-SUBNETS seq 20 permit 10.2.100.0/22 le 24
# P2P summarized
ip prefix-list HQ2-SUBNETS seq 25 permit 10.10.0.192/26 le 30

# Block tagged routes (Already advertised)
route-map HQ1-OSPF deny 10
match tag 34
match tag 12
exit

route-map HQ2-EIGRP permit 20
match ip address prefix-list HQ2-SUBNETS
set tag 34
exit

# Default route conditional injection
track 1 ip route 10.10.0.208 255.255.255.252 reachability
ip prefix-list DEFAULT_TRIGGER seq 5 permit 10.10.0.208/30

route-map INJECT_CON permit 10
match ip address prefix-list DEFAULT_TRIGGER
match track 1
exit

router ospf 100
router-id 10.10.0.13
network 10.10.0.116 0.0.0.3 area 0
redistribute eigrp 100 route-map HQ2-EIGRP
default-information originate metric 4000 route-map INJECT_CON

passive-interface default
no passive-interface g0/4
exit

int g0/4
ip ospf network point-to-point
end
